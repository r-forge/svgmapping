library(SVGMapping)


## Import sample data (taken from GEO acc:GSE7140 and SGD)

data(yeastExprData)

data(yeastAnnotMatrix)


mysvg <- loadSVG(system.file("extdata/example2.svg", package="SVGMapping"))


logratios <- yeastExprData

foldchange <- ifelse(logratios>=0, 2^(logratios), -2^-logratios)

mapDataSVG(mysvg,

           logratios,

           tooltipData=foldchange,

           mode="pie",

           annotation=yeastAnnotMatrix)


## Metabolites color stripes

mydata <- matrix(c(-2, -1.5, -1, -0.5, 2, 1.5, 1, 0.5), ncol=4,

                 byrow=TRUE,

                 dimnames=list(c("malate", "fumarate"), NULL))

mapDataSVG(mysvg, mydata, mode="fill")


## Erlenmeyers filling

mydata <- c(10, 25, 50)

names(mydata) <- c("citrate", "cis-aconitate", "isocitrate")

mapDataSVG(mysvg,

           numData=mydata, mode="fill",

           col=c("#FF0000","#00FF00","#0000FF"), colrange=range(mydata),

           tooltipData=paste(mydata, "mmol/L"))


mapDataSVG(mysvg, numData=mydata/100, mode="partial-fill")


## Show results in a browser

showSVG(mysvg)
